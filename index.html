function vowelIndices(word) {
  const vowels = 'aeiouyAEIOUY';
  let myArr = word.split('');
  let res = [];
  myArr.forEach((el, index) => {
    if (vowels.includes(el)) {
      res.push(index + 1);
    }
  })
  return res;
}
